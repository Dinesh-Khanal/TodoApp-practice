git command
1. check if git has already been installed
	git --version

2. to config git
	git config --global user.email "dineshkhanal@hotmail.com"
	git config --global user.name "Dinesh Khanal"
3. check user name and email
	git config --global user.name
	git config --global user.email
4. to add file to stage
	git add index.html
5. to see status
	git status
6. to remove from stage
	git rm --cached index.html
7. to add all html file
	git add *.html
8. to add all file
	git add .
9. to commit change
	git commit -m 'Initial commit'
10. to unstage all the staged files.
	git reset
11. to delete all local changes that have not been added to staging area
	git checkout
11. to clone remote repository
	git clone https://github.com/Dinesh-Khanal/TodoApp-practice.git
12. to pull the latest change from remote repository
	git pull <REMOTE> <name-of-branch>
13. View remote repositories
	git remote -v
15. to push all local commits
	git push <remote> <name of branch>
